\section{Referencial Teórico}
% ============================================================

\begin{frame}{Heurísticas Construtivas}
    \small
    \begin{block}{O que são?}
        Procedimentos aproximados que obtêm soluções de boa qualidade em
        tempo reduzido, \textbf{sem garantia de otimalidade}.
    \end{block}

    \bigskip
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \textbf{Características:}
            \begin{itemize}
                \item Constroem a solução em \emph{uma única passagem}
                \item Decisões locais guiadas por critério guloso
                \item Sem revisão das escolhas já feitas
                \item[$\Rightarrow$] Rápidas, porém limitadas em qualidade
            \end{itemize}
        \end{column}
        \begin{column}{0.42\textwidth}
            \textbf{Exemplo — mochila gulosa:}\\[0.2cm]
            Ordenar itens por $v_i/w_i$ decrescente.\\[0.2cm]
            \textit{Pode falhar:}\\[0.1cm]
            $W{=}10$, itens $(1,2)$ e $(10,15)$\\[0.1cm]
            Gulosa $\to$ valor $\mathbf{2}$\\[0.1cm]
            Ótimo $\to$ valor $\mathbf{15}$
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Meta-heurísticas}
    \small
    \begin{block}{O que são?}
        Métodos que coordenam busca local com estratégias de alto nível —
        capazes de \textbf{escapar de ótimos locais} e explorar
        robustamente o espaço de soluções.
    \end{block}

    \bigskip
    \begin{columns}
        \begin{column}{0.50\textwidth}
            \textbf{Conceitos centrais:}
            \begin{itemize}
                \item \textbf{Vizinhança} — soluções próximas
                \item \textbf{Busca local} — move-se para vizinho melhor
                \item \textbf{Intensificação} $\times$ \textbf{Diversificação}
            \end{itemize}
        \end{column}
        \begin{column}{0.46\textwidth}
            \textbf{Famílias principais:}
            \begin{itemize}
                \item \textit{Trajetória}: Busca Tabu, GRASP,\\
                      Simulated Annealing, VNS
                \item \textit{Populacionais}: Algoritmos Genéticos,
                      Colônia de Formigas
            \end{itemize}
        \end{column}
    \end{columns}

    \bigskip
    \begin{alertblock}{}
        Qualidade sensivelmente superior às heurísticas construtivas —
        ao custo de maior esforço computacional e calibração de parâmetros.
    \end{alertblock}
\end{frame}

\begin{frame}{Fronteira de Pareto — Definição}
    \small
    \begin{block}{Dominância de Pareto}
        $\mathbf{x}$ \textbf{domina} $\mathbf{y}$ se é \emph{não pior} em
        todos os objetivos \textbf{e} estritamente melhor em ao menos um.
        Soluções não dominadas formam a \textbf{Fronteira de Pareto}.
    \end{block}

    \bigskip
    \textbf{Neste trabalho:} dois objetivos a minimizar —
    \textit{gap} relativo $\times$ tempo (s)

    \medskip
    \begin{center}
        \begin{tabular}{clcc}

            \midrule

            $C_2$ & Pareto-ótima    & 4,0 & 1,2 \\
            $C_3$ & Pareto-ótima    & 1,5 & 4,0 \\
            \rowcolor{cinza}
            $C_4$ & Dominada por $C_2$ & 5,5 & 2,0 \\
            \rowcolor{cinza}
            $C_5$ & Dominada por $C_1$ & 9,0 & 1,0 \\
            \bottomrule
        \end{tabular}
    \end{center}
\end{frame}

\begin{frame}{Programação Fracionária Inteira e Dinkelbach}
    \small
    \begin{block}{Programação Fracionária Inteira (PFI)}
        Função objetivo é a \textbf{razão de dois lineares}:
        \[
            \max \quad \frac{f(x)}{g(x)}, \quad x \in S,\; g(x)>0
        \]
        Com variáveis inteiras $\Rightarrow$ \textbf{NP-difícil}.
        Não é linearizável diretamente para um solver MILP.
    \end{block}

    \bigskip
    \begin{block}{Método de Dinkelbach — referência para PFC II}
        Parametriza pelo subproblema
        $F(\lambda) = \max_{x \in S}\; f(x) - \lambda\, g(x)$
        e itera $\lambda_{k+1} = f(x_k)/g(x_k)$:
        \begin{itemize}
            \item Cada iteração resolve um \textbf{MILP ordinário}
            \item Convergência \textbf{superlinear} ao ótimo
            \item Estratégia de referência para o PSP em larga escala
        \end{itemize}
    \end{block}
\end{frame}

\section{Definição do Problema}
% ============================================================

\begin{frame}{O Problema de Seleção de Pedidos Ótima (PSP)}
    \small
    \textbf{Dados:} backlog de pedidos, estoque nos corredores,
    limites $[\mathrm{LB}, \mathrm{UB}]$

    \bigskip
    \textbf{Decidir simultaneamente:}
    \begin{enumerate}
        \item Subconjunto de pedidos $\mathcal{O}' \subseteq \mathcal{O}$
              que compõe a \textit{wave}
        \item Subconjunto de corredores $\mathcal{A}' \subseteq \mathcal{A}$
              a visitar
    \end{enumerate}

    \bigskip
    \textbf{Maximizar (Programação Fracionária Inteira — NP-difícil):}
    \[
        \max \quad
        \frac{\displaystyle\sum_{o \in \mathcal{O}'}\sum_{i \in \mathcal{I}_o} u_{oi}}
        {\displaystyle|\mathcal{A}'|}
    \]

    \smallskip
    \textbf{Restrições:}
    \begin{itemize}
        \item $\mathrm{LB} \leq \sum u_{oi}\,x_o \leq \mathrm{UB}$
              — tamanho mínimo e máximo da \textit{wave}
        \item Viabilidade física: os corredores visitados devem cobrir
              toda a demanda
    \end{itemize}
\end{frame}

\begin{frame}{Exemplo Numérico — Três Soluções Viáveis}
    \small
    Instância com 5 pedidos, 5 itens, 5 corredores;
    $\mathrm{LB}=5$, $\mathrm{UB}=12$

    \bigskip
    \begin{center}
        \begin{tabular}{c|c|c|c|c}
            \toprule
            \textbf{Solução} & $\mathcal{O}'$          & $\mathcal{A}'$
                             & \textbf{Unidades}       & \textbf{Objetivo}                       \\
            \midrule
            A                & $\{0,4\}$               & $\{0,1\}$         & 5  & $5/2 = 2{,}5$  \\
            B                & $\{0,2,3\}$             & $\{1,3,4\}$       & 12 & $12/3 = 4{,}0$ \\
            \rowcolor{verde!20}
            \textbf{C}       & $\{0,1,2,4\}$           & $\{1,3\}$         & 10
                             & $\mathbf{10/2 = 5{,}0}$                                           \\
            \bottomrule
        \end{tabular}
    \end{center}

    \bigskip
    \begin{alertblock}{Lição central}
        Mais pedidos \textbf{não} garante maior objetivo.
        A produtividade depende de \emph{concentrar} a demanda
        em \textbf{poucos corredores}.
        Solução C é ótima para esta instância.
    \end{alertblock}
\end{frame}

\section{Metodologia}
% ============================================================

\begin{frame}{Heurística Gulosa Parametrizável — Visão Geral}
    \small
    \begin{block}{Estrutura em duas etapas}
        \begin{enumerate}
            \item \textbf{Seleção de pedidos:} percorre a sequência
                  definida pelos parâmetros; aceita pedidos enquanto
                  respeita $\mathrm{UB}$ e o estoque agregado
            \item \textbf{Seleção de corredores:} dado o conjunto de
                  pedidos, determina os corredores que cobrem toda a
                  demanda
        \end{enumerate}
    \end{block}

    \bigskip
    \textbf{Parâmetros principais:}
    \begin{center}
        \begin{tabular}{ll}
            \toprule
            \texttt{order}        & critério de ordenação dos pedidos   \\
            \texttt{greedy}       & estratégia de seleção de corredores \\
            \texttt{first\_order} & pedido de referência (similaridade) \\
            \bottomrule
        \end{tabular}
    \end{center}
\end{frame}

\begin{frame}{Critérios de Ordenação de Pedidos (7)}
    \small
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{itemize}
                \item \texttt{desc} — decrescente por volume
                \item \texttt{asc} — crescente por volume
                \item \texttt{random} — aleatório
                \item \texttt{similar} — Jaccard crescente
                \item \texttt{diff} — Jaccard decrescente
                \item \texttt{similar\_weighted} — Jaccard pond.
                \item \texttt{diff\_weighted} — Jaccard pond.\ dec.
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Estratégias de corredores (3):}
            \begin{itemize}
                \item \texttt{simple} — pontuação estática
                \item \texttt{multi} — pontuação dinâmica
                      (recalculada a cada iteração)
                \item \texttt{exact} — CP-SAT (OR-Tools):
                      minimiza $|\mathcal{A}'|$ por PL inteira
            \end{itemize}
        \end{column}
    \end{columns}

    \bigskip
    \begin{center}
        \large
        $7 \times \underbrace{(1+4)}_{\text{sem/com first\_order}} \times 3
            \;-\;\text{combinações inválidas}
            = \mathbf{57}\ \text{variações}$
    \end{center}
\end{frame}

\section{Discussão e Limitações}
% ============================================================

\begin{frame}{Discussão dos Resultados}
    \small
    \begin{block}{Gaps de 0,69–0,78: patamar esperado}
        Heurísticas construtivas puras constroem a solução em
        uma única passagem — sem busca local ou refinamento iterativo.
        As soluções capturam $\approx$22–31\,\% da qualidade das
        MSC. Objetivo da Fase 1: \emph{filtrar comparativamente},
        não estabelecer benchmark absoluto.
    \end{block}

    \bigskip
    \begin{block}{Ordenação \texttt{desc} — principal fator de qualidade}
        Priorizar pedidos volumosos maximiza as unidades incorporadas
        à \textit{wave} nas primeiras iterações, quando o estoque
        ainda está intacto. Lógica gulosa naturalmente alinhada ao
        objetivo fracionário.
    \end{block}

    \bigskip
    \textbf{Limitações:}
    \begin{itemize}
        \item Resultados restritos ao \textit{dataset} A (menor porte)
        \item Sem testes de significância estatística formal
        \item Apenas heurísticas construtivas — sem busca local
    \end{itemize}
\end{frame}

\section{Conclusão e Próximos Passos}
% ============================================================

\begin{frame}{Conclusão — PFC I}
    \small
    \textbf{Objetivos cumpridos integralmente:}
    \begin{enumerate}
        \item Formalização do PSP e enquadramento na literatura do OBP
        \item Implementação da heurística gulosa com \textbf{57 variações}
              (7 critérios $\times$ 3 estratégias de corredores)
        \item Identificação das configurações Pareto-ótimas:
              \textbf{A4, A5 e A6}
    \end{enumerate}

    \bigskip
    \begin{block}{Resultado central}
        A ordenação decrescente (\texttt{desc}) é o fator dominante
        na qualidade. A estratégia de corredores opera um
        \textbf{\textit{trade-off} controlável}: de A6 (rápido) a
        A4 (melhor qualidade), todos não dominados pela fronteira
        de Pareto.
